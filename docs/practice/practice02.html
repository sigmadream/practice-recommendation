def predict(df, year):
    X = np.array(range(2012, 2023))
    X = sm.add_constant(X)
    n, m = df.shape
    l = []
    for i in range(0, n):
        y = df.iloc[i, :].values
        est = sm.OLS(y, X).fit()
        l.append(float(est.predict(np.array([1, year]))))
    return pd.Series(l, index=df.index)
